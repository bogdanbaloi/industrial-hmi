# AddressSanitizer suppression file
# Suppress known leaks in third-party libraries (GTK, Pango, Fontconfig)
# These are not bugs in our code - they are known issues in system libraries.

# Fontconfig caches font data globally and never frees it
leak:libfontconfig
leak:FcValueSave
leak:FcPatternObjectInsertElt

# Pango caches layout/font data globally
leak:libpango
leak:pango_layout
leak:pango_font

# GLib/GObject type system registers types permanently
leak:libglib
leak:libgobject
leak:g_type_register
leak:g_type_class_ref

# GTK global state
leak:libgtk-4
leak:gtk_init
leak:gtk_css
